\section{Cálculo de raíces}

\begin{equation}
f(x) = x^3-\ln(x)-3
\label{eq:calc:f}
\end{equation}

\begin{equation}
a = 1
\label{eq:calc:a}
\end{equation}

\begin{equation}
b = 2
\label{eq:calc:b}
\end{equation}

\begin{algorithm}
\caption{Método de dicotomía}
\label{alg:calc:dicotomia}
\begin{algorithmic}
\STATE $tol = 1e-6$
\WHILE{$b - a > tol$}
    \STATE $c = (a + b)/2$
    \IF{$f(a) \cdot f(c) < 0$}
        \STATE $b = c$
    \ELSE
        \STATE $a = c$
    \ENDIF
\ENDWHILE
\RETURN $c$
\end{algorithmic}
\end{algorithm}
