\begin{algorithm}
\label{alg:calc:Ejemplo con ELSIF}
\begin{algorithmic}
        \STATE $i = 10$
        \STATE $j = \sqrt{i^2 + 5}$
        \IF{$i \ge 5$}
            \STATE $i = i - 1$
            \WHILE{$j \le 20$}
                \STATE $j = j + \log(i)$
            \ENDWHILE
        \ELSIF{$i \le 3$}
            \STATE $i = i + 2$
        \ENDIF
        \RETURN $j$
\end{algorithmic}
\end{algorithm}
